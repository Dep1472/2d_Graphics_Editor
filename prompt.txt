#define ROWS        20

struct Object {

    int  x, y;

    int  active;
};

char          canvas[ROWS][COLS];   
struct Object objects[MAX_OBJECTS]; 
int           obj_count = 0;        

void clear_canvas(void)
{
    int r, c;
    for (r = 0; r < ROWS; r++)
        for (c = 0; c < COLS; c++)
            canvas[r][c] = BLANK;
}


void set_cell(int x, int y, char ch)
{
    if (x >= 0 && x < COLS && y >= 0 && y < ROWS)
        canvas[y][x] = ch;
}

void display_picture(void)
{
    int r, c;

    printf("+");
    for (c = 0; c < COLS; c++) printf("-");
    printf("+\n");

    printf("+\n");
}


void draw_rectangle(int x, int y, int w, int h, char ch)
{
    int i, j;

    for (i = 0; i < w; i++) {
        set_cell(x + i, y,         ch);
        set_cell(x + i, y + h - 1, ch);
    }

    for (j = 0; j < h; j++) {
        set_cell(x,         y + j, ch);
        set_cell(x + w - 1, y + j, ch);
    }
}

void draw_circle(int cx, int cy, int r, char ch)
{
    int x, y;
    double dx, dy, dist;

    for (y = -r; y <= r; y++) {
        for (x = -(r * 2); x <= (r * 2); x++) {
            dx   = x / 2.0;
            dy   = (double)y;
            dist = sqrt(dx * dx + dy * dy);
            if (fabs(dist - r) < 0.6)
                set_cell(cx + x, cy + y, ch);
        }
    }
}

void draw_triangle(int x, int y, int w, int h, char ch)
{
    int row, half, apex, left, right, i;

    apex = w / 2;  

    for (row = 0; row < h; row++) {
        half  = (int)round((double)row / (double)(h - 1) * apex);
        left  = x + apex - half;
        right = x + apex + half;

        if (row == 0) {
            set_cell(x + apex, y, ch);         
        } else if (row == h - 1) {
            for (i = left; i <= right; i++)     
                set_cell(i, y + row, ch);
        } else {
            set_cell(left,  y + row, ch);      
            set_cell(right, y + row, ch);     
        }
    }
}


void rebuild_canvas(void)
{
    int i;
    clear_canvas();

    for (i = 0; i < obj_count; i++) {
        if (!objects[i].active) continue;

        switch (objects[i].type) {
            case SHAPE_RECT:
                draw_rectangle(objects[i].x, objects[i].y,
                               objects[i].w, objects[i].h,
                               objects[i].ch);
                break;

            case SHAPE_CIRCLE:
                draw_circle(objects[i].x, objects[i].y,
                            objects[i].r, objects[i].ch);
                break;

            case SHAPE_TRIANGLE:
                draw_triangle(objects[i].x, objects[i].y,
                              objects[i].w, objects[i].h,
                              objects[i].ch);
                break;
        }
    }
}


int add_object(struct Object obj)
{
    if (obj_count >= MAX_OBJECTS) {
        printf("Canvas is full — cannot add more objects.\n");
        return -1;
    }
    obj.active          = 1;
    objects[obj_count]  = obj;
    obj_count++;
    rebuild_canvas();
    printf("Object added at index %d.\n", obj_count - 1);
    return obj_count - 1;
}

void delete_object(int i)
{
    if (i < 0 || i >= obj_count || !objects[i].active) {
        printf("Invalid index — nothing deleted.\n");
        return;
    }
    objects[i].active = 0;
    rebuild_canvas();
    printf("Object %d deleted.\n", i);
}

void modify_object(int i, struct Object new_obj)
{
    if (i < 0 || i >= obj_count) {
        printf("Invalid index — nothing modified.\n");
        return;
    }
    new_obj.active = 1;
    objects[i]     = new_obj;
    rebuild_canvas();
    printf("Object %d modified.\n", i);
}


void list_objects(void)
{
    int   i, found = 0;
    char *names[] = { "", "Rectangle", "Circle", "Triangle" };

    printf("\n--- Active Objects ---\n");

    printf("----------------------\n\n");
}
